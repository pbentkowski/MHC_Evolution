50 void Antigen::calculateEpitopes(unsigned long mhcSize){
51  unsigned long vecSize = TheAntigen.size() - mhcSize;
52  boost::dynamic_bitset<> bitEpitope(mhcSize);
53  longIntVec tmpEpis(vecSize);
54  unsigned long tmpEpisSize = tmpEpis.size();
55  for(unsigned long i = 0; i < tmpEpisSize; ++i){
56  for(unsigned long j = i; j < (i + mhcSize); ++j){
57  bitEpitope[j-i] = TheAntigen[j];
58  }
59  tmpEpis[i] = bitEpitope.to_ulong();
60  }
61  Epitopes = tmpEpis;
62 }
