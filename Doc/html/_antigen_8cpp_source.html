51 void Antigen::calculateEpitopes(int mhcSize){
52  int vecSize = TheAntigen.size() - mhcSize;
53  boost::dynamic_bitset<> bitEpitope(mhcSize);
54  longIntVec tmpEpis(vecSize);
55  for(int i = 0; i < tmpEpis.size(); ++i){
56  for(int j = i; j < (i + mhcSize); ++j){
57  bitEpitope[j-i] = TheAntigen[j];
58  }
59  tmpEpis[i] = bitEpitope.to_ulong();
60  }
61  Epitopes = tmpEpis;
62 }

103 void Antigen::mutateAntigenBitByBit(double pm_mut_probabl, int mhcSize, int timeStamp){
104  boost::dynamic_bitset<> bitgene;
105  bitgene = TheAntigen;
106  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
107  for(boost::dynamic_bitset<>::size_type i = 0; i < bitgene.size(); ++i) {
108  if(p_RandomNumbs->NextReal(0.0, 1.0) < pm_mut_probabl) {
109  bitgene[i].flip();
110  }
111  }
112  if(TheAntigen != bitgene){
113  ParentTags.push_back(AntigenTag);
114  MutationTime.push_back(timeOfOrigin);
115  Tagging_system* pTagging_system = Tagging_system::getInstance();
116  AntigenTag = pTagging_system->getTag();
