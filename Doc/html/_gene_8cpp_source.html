172 void Gene::mutateGeneBitByBit(double pm_mut_probabl, int timeStamp) {
173  int currentGene = TheGene;
174  boost::dynamic_bitset<> bitgene(BitStringLength, TheGene);
175  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
176  for(boost::dynamic_bitset<>::size_type i = 0; i < bitgene.size(); ++i) {
177  if(p_RandomNumbs->NextReal(0.0, 1.0) < pm_mut_probabl) {
178  bitgene[i].flip();
179  }
180  TheGene = (int) bitgene.to_ulong();
181  }
182  if(currentGene != TheGene){
183  ParentTags.push_back(GenesTag);
184  MutationTime.push_back(timeOfOrigin);
185  Tagging_system* pTagging_system = Tagging_system::getInstance();
186  GenesTag = pTagging_system->getTag();
187  timeOfOrigin = timeStamp;
188  }
189 }
