52 auto getShannonIndx(std::vector<unsigned long> GeneVals){
53  int Types = 0;
54  double Summ = 0.0;
55  double tot_gene_numb = (double) GeneVals.size();
56  if(GeneVals.size()){
57  unsigned long GeneCounter = GeneVals.size();
58  bool IfCountedLyst[GeneCounter];
59  for (unsigned long w = 0; w < GeneCounter; ++w) {
60  IfCountedLyst[w] = false;
61  }
62  double abundance;
63  for (unsigned long i = 0; i < GeneCounter; ++i) {
64  if (!IfCountedLyst[i]) {
65  Types += 1;
66  abundance = 1.0;
67  for (unsigned long j = i + 1; j < GeneCounter; ++j) {
68  if (i != j && !IfCountedLyst[j] && GeneVals[i] == GeneVals[j]){
69  abundance += 1.0;
70  IfCountedLyst[j] = true;
71  }
72  }
73  // the proper calculations
74  Summ = Summ - ((abundance / tot_gene_numb)
75  * std::log(abundance / tot_gene_numb));
76  }
77  }
78  }
79  // ShannIndx (double), typesOfGenes (int), numberOfAllGenes (double)
80  auto RTRN = std::make_tuple( Summ, Types, tot_gene_numb );
81  return RTRN;
82 }

142 bool DataHandler::checkParamsIfWrong(unsigned int numberOfThreads, unsigned long geneLength, unsigned long antigenLength,
143  int hostPopSize, int pathoPopSize, int patho_sp, unsigned long hostGeneNumbb,
144  int patoPerHostGeneration, int numOfHostGenerations, double hostMutationProb,
145  double pathoMutationProb, int HeteroHomo, double hostDeletion, double hostDuplication,
146  unsigned long maxGene, double alpha, int numberOfMates){
147  bool ifError = false;
148  if (numberOfThreads < 0){
149  std::cout << "\nError in the number of threads. It has to be a positive integer!." << std::endl;
150  ifError = true;
151  }
152  if (geneLength > 31){
153  std::cout << "\nError in number of bits per gene. ";
154  std::cout << geneLength << " bits in genes is bit too much. ";
155  std::cout << "Try something less radical, e.g. smaller than 31." << std::endl;
156  ifError = true;
157  }
158  if (hostMutationProb < 0.0 or hostMutationProb > 1.0){
159  std::cout << "\nError in the hosts' mutation probability. It has to be " <<

161  ifError = true;
162  }
163  if (hostDeletion < 0.0 or hostDeletion > 1.0){
164  std::cout << "\nError in the hosts' probability of deletion of a gene. " <<
165  "It has to be within the range [0, 1]." << std::endl;
166  ifError = true;
167  }
168  if (hostDuplication < 0.0 or hostDuplication > 1.0){
169  std::cout << "\nError in the hosts' duplication of a gene probability. " <<
170  "It has to be within the range [0, 1]." << std::endl;
171  ifError = true;
172  }
173  if (HeteroHomo != 10 and HeteroHomo != 11){
174  std::cout << "\nError in the heterozygote advantage / lack of advantage " <<
175  "mode. It has to be 10 for heterozygote advantage or 11 for " <<
176  "lack of thereof." << std::endl;
177  ifError = true;
178  }
179  if (maxGene < 1.0 or maxGene < hostGeneNumbb){
180  std::cout << "\nError in the hosts' maximal number of genes per " <<
181  "chromosome. It has to be at least one, but not less then the " <<
182  "number used to initialize the system. "<< std::endl;
183  ifError = true;
184  }
185  if (numberOfMates < 0 or numberOfMates > hostPopSize){
186  std::cout << "\nError in the hosts' number of sexual partners. It has " <<
187  "to be larger than 0 but not more than host population size ." << std::endl;
188  ifError = true;
189  }
190  if (alpha < 0.0 or alpha > 1.0){
191  std::cout << "\nError in the hosts' alpha factor for the host fitness function. " <<
192  "It has to be within the range [0, 1]." << std::endl;
193  ifError = true;
194  }
195  return ifError;
196 }

342 void DataHandler::saveHostGeneticDivers(Environment& EnvObj, int tayme){
343  if(ifFirstHostGeneDivRun){
344  std::ofstream HostGeneDivFile;
345  HostGeneDivFile.open("HostsGeneDivers.csv");
346  HostGeneDivFile << "#time pop_size tot_num_of_genes num_of_MHC_types" <<
347  " Shannon_indx mean_fitness std_fitness" << std::endl;
348  HostGeneDivFile.close();
349  ifFirstHostGeneDivRun = false;
350  }
351  std::vector<unsigned long> AllTheGeneVals;
352  std::vector<double> Fitness;
353  double popSize = (double) EnvObj.getHostsPopSize();
354 // int homoLociNum = -1; // not applicable at the moment!
355 // double tot_gene_numb = 0;
356  Fitness.clear();
357  for(unsigned long i = 0; i < EnvObj.getHostsPopSize(); ++i){
358 // tot_gene_numb += EnvObj.getSingleHostGenomeSize(i);
359  Fitness.push_back(EnvObj.getHostFitness(i));
360  // Harvesting genes from Chromosome One in all hosts
361  for(unsigned long j = 0; j < EnvObj.getSingleHostChromoOneSize(i); ++j){
362  AllTheGeneVals.push_back(EnvObj.getSingleHostRealGeneOne(i, j));
363  }
364  // Harvesting genes from Chromosome Two in all Phosts
365  for(unsigned long m = 0; m < EnvObj.getSingleHostChromoTwoSize(i); ++m){
366  AllTheGeneVals.push_back(EnvObj.getSingleHostRealGeneTwo(i, m));
367  }
368  }
369  // Calculating the Shannon index et al. plus extracting it from the tuple
370  auto ShOut = getShannonIndx(AllTheGeneVals);
371  double Summ = std::get < 0 >( ShOut );
372  int mhcTypes = std::get < 1 >( ShOut );
373  double tot_gene_numb = std::get < 2 >( ShOut );
374 
375  // calculating coefficient of variation of hosts fitness
376  double cv_sum = std::accumulate(Fitness.begin(), Fitness.end(), 0.0);
377  double cv_mean = cv_sum / Fitness.size();
378  double sq_sum = std::inner_product(Fitness.begin(), Fitness.end(),
379  Fitness.begin(), 0.0);
380  double stdev = std::sqrt(sq_sum / Fitness.size() - cv_mean * cv_mean);
381 
382  std::ofstream HostGenomesFile;
383  HostGenomesFile.open("HostsGeneDivers.csv",
384  std::ios::out | std::ios::ate | std::ios::app);
385  HostGenomesFile << tayme << " " << popSize << " " << tot_gene_numb << " " <<
386  mhcTypes << " " << Summ << " " << cv_mean <<
387  " " << stdev << std::endl;
388  HostGenomesFile.close();
389 }
