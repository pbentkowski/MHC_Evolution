50 auto getShannonIndx(std::vector<unsigned long> GeneVals){
51  int Types = 0;
52  double Summ = 0.0;
53  double tot_gene_numb = (double) GeneVals.size();
54  if(GeneVals.size()){
55  unsigned long GeneCounter = GeneVals.size();
56  bool IfCountedLyst[GeneCounter];
57  for (unsigned long w = 0; w < GeneCounter; ++w) {
58  IfCountedLyst[w] = false;
59  }
60  double abundance;
61  for (unsigned long i = 0; i < GeneCounter; ++i) {
62  if (!IfCountedLyst[i]) {
63  Types += 1;
64  abundance = 1.0;
65  for (unsigned long j = i + 1; j < GeneCounter; ++j) {
66  if (i != j && !IfCountedLyst[j] && GeneVals[i] == GeneVals[j]){
67  abundance += 1.0;
68  IfCountedLyst[j] = true;
69  }
70  }
71  // the proper calculations
72  Summ = Summ - ((abundance / tot_gene_numb)
73  * std::log(abundance / tot_gene_numb));
74  }
75  }
76  }
77  // ShannIndx (double), typesOfGenes (int), numberOfAllGenes (double)
78  auto RTRN = std::make_tuple( Summ, Types, tot_gene_numb );
79  return RTRN;
80 }
