299 void Environment::selectAndReprodHostsAddOffspring(){
300  unsigned pop_size = HostPopulation.size();
301  double sum_of_fit = 0;
302  double rnd;
303  for(unsigned i = 0; i < pop_size; ++i){
304  sum_of_fit += HostPopulation[i].getFitness();
305  }
306  if(sum_of_fit == 0){
307  return;
308  }
309  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
310  int n = 0;
311  aley_oop:
312  while(n < pop_size){
313  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
314  for(unsigned k = 0; k < pop_size; ++k) {
315  rnd = rnd - HostPopulation[k].getFitness();
316  if(rnd <= 0){
317  HostPopulation[k].SelectedForReproduction += 1;
318  n += 1;
319  goto aley_oop;
320  }
321  }
322  }
323  // Remove the unfit ones
324  int PopSize = HostPopulation.size();
325  for (int i = PopSize - 1; i >= 0; --i){
326  if (HostPopulation[i].SelectedForReproduction == 0){
327  HostPopulation.erase(HostPopulation.begin() + i);
328  }
329  }
330  // Reproduce the fit ones
331 // PopSize = HostPopulation.size();
332 // for (int i = PopSize - 1; i >= 0; --i){
333 // if (HostPopulation[i].SelectedForReproduction){
334 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
335 // HostPopulation.push_back(HostPopulation[k]);
336 // }
337 // }
338 // }
339  // Random mating
340  PopSize = HostPopulation.size();
341  int emptySlots = pop_size - PopSize;
342  int indx_father, indx_mother;
343  for (int p = 0; p < emptySlots; ++p){
344  indx_father = p_RandomNumbs->NextInt(0, PopSize);
345  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
346  while (indx_father == indx_mother){
347  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
348  }
349  HostPopulation.push_back(HostPopulation[indx_mother]);
350  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
351  // Randomly swaps places of chromosomes to avoid situation when they
