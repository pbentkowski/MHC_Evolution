566 void Environment::selectAndReprodHostsAddOffspring(){
567  unsigned pop_size = HostPopulation.size();
568  double sum_of_fit = 0;
569  double rnd;
570  for(unsigned i = 0; i < pop_size; ++i){
571  sum_of_fit += HostPopulation[i].getFitness();
572  }
573  if(sum_of_fit == 0){
574  return;
575  }
576  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
577  int n = 0;
578  aley_oop:
579  while(n < pop_size){
580  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
581  for(unsigned k = 0; k < pop_size; ++k) {
582  rnd = rnd - HostPopulation[k].getFitness();
583  if(rnd <= 0){
584  HostPopulation[k].SelectedForReproduction += 1;
585  n += 1;
586  goto aley_oop;
587  }
588  }
589  }
590  // Remove the unfit ones
591  int PopSize = HostPopulation.size();
592  for (int i = PopSize - 1; i >= 0; --i){
593  if (HostPopulation[i].SelectedForReproduction == 0){
594  HostPopulation.erase(HostPopulation.begin() + i);
595  }
596  }
597  // Reproduce the fit ones
598 // PopSize = HostPopulation.size();
599 // for (int i = PopSize - 1; i >= 0; --i){
600 // if (HostPopulation[i].SelectedForReproduction){
601 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
602 // HostPopulation.push_back(HostPopulation[k]);
603 // }
604 // }
605 // }
606  // Random mating
607  PopSize = HostPopulation.size();
608  int emptySlots = pop_size - PopSize;
609  int indx_father, indx_mother;
610  for (int p = 0; p < emptySlots; ++p){
611  indx_father = p_RandomNumbs->NextInt(0, PopSize);
612  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
613  while (indx_father == indx_mother){
614  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
615  }
616  HostPopulation.push_back(HostPopulation[indx_mother]);
617  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
618  // Randomly swaps places of chromosomes to avoid situation when they
619  // effectively become two separate populations.
620  HostPopulation.back().swapChromosomes();
621  }
