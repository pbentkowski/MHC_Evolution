493 void Environment::selectAndReprodHostsNoMating() {
494  std::vector<Host> NewHostsVec;
495  NewHostsVec.clear();
496  unsigned long pop_size = HostPopulation.size();
497  double sum_of_fit = 0;
498  double rnd;
499  for(unsigned long i = 0; i < pop_size; ++i){
500  sum_of_fit += HostPopulation[i].getFitness();
501  }
502  if(sum_of_fit == 0){
503  return;
504  }
505  Random * rngGenPtr = mRandGenArr;
506  #pragma omp single
507  {
508  int n = 0;
509  aley_oop:
510  while (n < pop_size) {
511  rnd = rngGenPtr[omp_get_thread_num()].getRealDouble(0, sum_of_fit);
512  unsigned long HostPopulationSize = HostPopulation.size();
513  for (unsigned long k = 0; k < HostPopulationSize; ++k) {
514  rnd = rnd - HostPopulation[k].getFitness();
515  if (rnd <= 0) {
516  HostPopulation[k].SelectedForReproduction += 1;
517  NewHostsVec.push_back(HostPopulation[k]);
518  n += 1;
519  goto aley_oop;
520  }
521  }
522  }
523  }
524  if (HostPopulation.size() == NewHostsVec.size()){
525  HostPopulation.clear();
526  HostPopulation = NewHostsVec;
527  }else{
528  std::cout << "Error in selectAndReprodHostsNoMating(): Size mismatch " <<
529  "between the new and the old population!" << std::endl;
530  std::cout << "old pop: " << HostPopulation.size() <<
531  " | new pop: " << NewHostsVec.size() << std::endl;
532  }
533 }
