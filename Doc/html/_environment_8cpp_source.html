110 void Environment::setPathoPopulationSeparateGenePools(int pop_size, int gene_size,
111  int chrom_size, int numb_of_species, int timeStamp){
112  if (numb_of_species > pop_size) numb_of_species = pop_size;
113  int step_of_gene = ((int) std::pow(2, gene_size) -1 ) / numb_of_species;
114  int indiv_per_species = pop_size / numb_of_species;
115  int indiv_left = pop_size % numb_of_species;
116  std::vector<Pathogen> OneSpeciesVector;
117  for (int i = 0; i < numb_of_species; ++i){
118  for(int j = 0; j < indiv_per_species; ++j){

120  if(i+1 != indiv_per_species){
121  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
122  i*step_of_gene, (i+1)*step_of_gene - 1, timeStamp);
123  } else {
124  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
125  i*step_of_gene, (int) (std::pow(2, gene_size) - 1), timeStamp);
126  }
127  if(indiv_left){
128  OneSpeciesVector.push_back(Pathogen());
129  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
130  i*step_of_gene, (i+1)*step_of_gene - 1, timeStamp);
131  indiv_left--;
132  }
133  }
134  PathPopulation.push_back(OneSpeciesVector);
135  OneSpeciesVector.clear();

352 void Environment::selectAndReprodHostsAddOffspring(){
353  unsigned pop_size = HostPopulation.size();
354  double sum_of_fit = 0;
355  double rnd;
356  for(unsigned i = 0; i < pop_size; ++i){
357  sum_of_fit += HostPopulation[i].getFitness();
358  }
359  if(sum_of_fit == 0){
360  return;
361  }
362  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
363  int n = 0;
364  aley_oop:
365  while(n < pop_size){
366  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
367  for(unsigned k = 0; k < pop_size; ++k) {
368  rnd = rnd - HostPopulation[k].getFitness();
369  if(rnd <= 0){
370  HostPopulation[k].SelectedForReproduction += 1;
371  n += 1;
372  goto aley_oop;
373  }
374  }
375  }
376  // Remove the unfit ones
377  int PopSize = HostPopulation.size();
378  for (int i = PopSize - 1; i >= 0; --i){
379  if (HostPopulation[i].SelectedForReproduction == 0){
380  HostPopulation.erase(HostPopulation.begin() + i);
381  }
382  }
383  // Reproduce the fit ones
384 // PopSize = HostPopulation.size();
385 // for (int i = PopSize - 1; i >= 0; --i){
386 // if (HostPopulation[i].SelectedForReproduction){
387 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
388 // HostPopulation.push_back(HostPopulation[k]);
389 // }
390 // }
391 // }
392  // Random mating
393  PopSize = HostPopulation.size();
394  int emptySlots = pop_size - PopSize;
395  int indx_father, indx_mother;
396  for (int p = 0; p < emptySlots; ++p){
397  indx_father = p_RandomNumbs->NextInt(0, PopSize);
398  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
399  while (indx_father == indx_mother){
400  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
401  }
402  HostPopulation.push_back(HostPopulation[indx_mother]);
403  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
404  // Randomly swaps places of chromosomes to avoid situation when they
405  // effectively become two separate populations.
406  HostPopulation.back().swapChromosomes();
407  }
