112 void Environment::setPathoPopulationSeparateGenePools(int pop_size, int gene_size,
113  int chrom_size, int numb_of_species, int timeStamp){
114  if (numb_of_species > pop_size) numb_of_species = pop_size;
115  int step_of_gene = ((int) std::pow(2, gene_size) -1 ) / numb_of_species;
116  int indiv_per_species = pop_size / numb_of_species;
117  int indiv_left = pop_size % numb_of_species;
118  std::vector<Pathogen> OneSpeciesVector;
119  for (int i = 0; i < numb_of_species; ++i){
120  for(int j = 0; j < indiv_per_species; ++j){
121  OneSpeciesVector.push_back(Pathogen());
122  if(i+1 != indiv_per_species){
123  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
124  i*step_of_gene, (i+1)*step_of_gene - 1, timeStamp);
125  } else {
126  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
127  i*step_of_gene, (int) (std::pow(2, gene_size) - 1), timeStamp);
128  }
129  if(indiv_left){
130  OneSpeciesVector.push_back(Pathogen());
131  OneSpeciesVector.back().setNewPathogen(chrom_size, gene_size, i,
132  i*step_of_gene, (i+1)*step_of_gene - 1, timeStamp);
133  indiv_left--;
134  }
135  }
136  PathPopulation.push_back(OneSpeciesVector);
137  OneSpeciesVector.clear();
138  }

312 void Environment::selectAndReprodHostsAddOffspring(){
313  unsigned pop_size = HostPopulation.size();
314  double sum_of_fit = 0;
315  double rnd;
316  for(unsigned i = 0; i < pop_size; ++i){
317  sum_of_fit += HostPopulation[i].getFitness();
318  }
319  if(sum_of_fit == 0){
320  return;
321  }
322  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
323  int n = 0;
324  aley_oop:
325  while(n < pop_size){
326  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
327  for(unsigned k = 0; k < pop_size; ++k) {
328  rnd = rnd - HostPopulation[k].getFitness();
329  if(rnd <= 0){
330  HostPopulation[k].SelectedForReproduction += 1;
331  n += 1;
332  goto aley_oop;
333  }
334  }
335  }
336  // Remove the unfit ones
337  int PopSize = HostPopulation.size();
338  for (int i = PopSize - 1; i >= 0; --i){
339  if (HostPopulation[i].SelectedForReproduction == 0){
340  HostPopulation.erase(HostPopulation.begin() + i);
341  }
342  }
343  // Reproduce the fit ones
344 // PopSize = HostPopulation.size();
345 // for (int i = PopSize - 1; i >= 0; --i){
346 // if (HostPopulation[i].SelectedForReproduction){
347 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
348 // HostPopulation.push_back(HostPopulation[k]);
349 // }
350 // }
351 // }
352  // Random mating
353  PopSize = HostPopulation.size();
354  int emptySlots = pop_size - PopSize;
355  int indx_father, indx_mother;
356  for (int p = 0; p < emptySlots; ++p){
357  indx_father = p_RandomNumbs->NextInt(0, PopSize);
358  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
359  while (indx_father == indx_mother){
360  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
361  }
362  HostPopulation.push_back(HostPopulation[indx_mother]);
363  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
364  // Randomly swaps places of chromosomes to avoid situation when they
