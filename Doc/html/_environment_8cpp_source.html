559 void Environment::selectAndReprodHostsAddOffspring(){
560  unsigned pop_size = HostPopulation.size();
561  double sum_of_fit = 0;
562  double rnd;
563  for(unsigned i = 0; i < pop_size; ++i){
564  sum_of_fit += HostPopulation[i].getFitness();
565  }
566  if(sum_of_fit == 0){
567  return;
568  }
569  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
570  int n = 0;
571  aley_oop:
572  while(n < pop_size){
573  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
574  for(unsigned k = 0; k < pop_size; ++k) {
575  rnd = rnd - HostPopulation[k].getFitness();
576  if(rnd <= 0){
577  HostPopulation[k].SelectedForReproduction += 1;
578  n += 1;
579  goto aley_oop;
580  }
581  }
582  }
583  // Remove the unfit ones
584  int PopSize = HostPopulation.size();
585  for (int i = PopSize - 1; i >= 0; --i){
586  if (HostPopulation[i].SelectedForReproduction == 0){
587  HostPopulation.erase(HostPopulation.begin() + i);
588  }
589  }
590  // Reproduce the fit ones
591 // PopSize = HostPopulation.size();
592 // for (int i = PopSize - 1; i >= 0; --i){
593 // if (HostPopulation[i].SelectedForReproduction){
594 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
595 // HostPopulation.push_back(HostPopulation[k]);
596 // }
597 // }
598 // }
599  // Random mating
600  PopSize = HostPopulation.size();
601  int emptySlots = pop_size - PopSize;
602  int indx_father, indx_mother;
603  for (int p = 0; p < emptySlots; ++p){
604  indx_father = p_RandomNumbs->NextInt(0, PopSize);
605  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
606  while (indx_father == indx_mother){
607  indx_mother = p_RandomNumbs->NextInt(0, PopSize);

611  // Randomly swaps places of chromosomes to avoid situation when they
