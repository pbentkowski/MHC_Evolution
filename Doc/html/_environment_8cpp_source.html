338 void Environment::selectAndReprodHostsAddOffspring(){
339  unsigned pop_size = HostPopulation.size();
340  double sum_of_fit = 0;
341  double rnd;
342  for(unsigned i = 0; i < pop_size; ++i){
343  sum_of_fit += HostPopulation[i].getFitness();
344  }
345  if(sum_of_fit == 0){
346  return;
347  }
348  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
349  int n = 0;
350  aley_oop:
351  while(n < pop_size){
352  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
353  for(unsigned k = 0; k < pop_size; ++k) {
354  rnd = rnd - HostPopulation[k].getFitness();
355  if(rnd <= 0){
356  HostPopulation[k].SelectedForReproduction += 1;
357  n += 1;
358  goto aley_oop;
359  }
360  }
361  }
362  // Remove the unfit ones
363  int PopSize = HostPopulation.size();
364  for (int i = PopSize - 1; i >= 0; --i){
365  if (HostPopulation[i].SelectedForReproduction == 0){
366  HostPopulation.erase(HostPopulation.begin() + i);
367  }
368  }
369  // Reproduce the fit ones
370 // PopSize = HostPopulation.size();
371 // for (int i = PopSize - 1; i >= 0; --i){
372 // if (HostPopulation[i].SelectedForReproduction){
373 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
374 // HostPopulation.push_back(HostPopulation[k]);
375 // }
376 // }
377 // }
378  // Random mating
379  PopSize = HostPopulation.size();
380  int emptySlots = pop_size - PopSize;
381  int indx_father, indx_mother;
382  for (int p = 0; p < emptySlots; ++p){
383  indx_father = p_RandomNumbs->NextInt(0, PopSize);
384  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
385  while (indx_father == indx_mother){
386  indx_mother = p_RandomNumbs->NextInt(0, PopSize);
387  }
388  HostPopulation.push_back(HostPopulation[indx_mother]);
389  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
390  // Randomly swaps places of chromosomes to avoid situation when they
391  // effectively become two separate populations.
392  HostPopulation.back().swapChromosomes();
393  }
