560 void Environment::selectAndReprodHostsAddOffspring(){
561  unsigned long pop_size = HostPopulation.size();
562  double sum_of_fit = 0;
563  double rnd;
564  for(unsigned i = 0; i < pop_size; ++i){
565  sum_of_fit += HostPopulation[i].getFitness();
566  }
567  if(sum_of_fit == 0){
568  return;
569  }
570  RandomNumbs * p_RandomNumbs = RandomNumbs::getInstance();
571  int n = 0;
572  aley_oop:
573  while(n < pop_size){
574  rnd = p_RandomNumbs->NextReal(0, sum_of_fit);
575  for(unsigned k = 0; k < pop_size; ++k) {
576  rnd = rnd - HostPopulation[k].getFitness();
577  if(rnd <= 0){
578  HostPopulation[k].SelectedForReproduction += 1;
579  n += 1;
580  goto aley_oop;
581  }
582  }
583  }
584  // Remove the unfit ones
585  unsigned long PopSize = HostPopulation.size();
586  for (unsigned long i = PopSize - 1; i >= 0; --i){
587  if (HostPopulation[i].SelectedForReproduction == 0){
588  HostPopulation.erase(HostPopulation.begin() + i);
589  }
590  }
591  // Reproduce the fit ones
592 // PopSize = HostPopulation.size();
593 // for (int i = PopSize - 1; i >= 0; --i){
594 // if (HostPopulation[i].SelectedForReproduction){
595 // for(int k = 1; k < HostPopulation[i].SelectedForReproduction; ++k){
596 // HostPopulation.push_back(HostPopulation[k]);
597 // }
598 // }
599 // }
600  // Random mating
601  PopSize = HostPopulation.size();
602  unsigned long emptySlots = pop_size - PopSize;
603  unsigned long indx_father, indx_mother;
604  for (unsigned long p = 0; p < emptySlots; ++p){
605  indx_father = p_RandomNumbs->NextLongInt(0, PopSize);
606  indx_mother = p_RandomNumbs->NextLongInt(0, PopSize);
607  while (indx_father == indx_mother){
608  indx_mother = p_RandomNumbs->NextLongInt(0, PopSize);
609  }
610  HostPopulation.push_back(HostPopulation[indx_mother]);
611  HostPopulation.back().assignChromTwo(HostPopulation[indx_father].getChromosomeTwo());
612  // Randomly swaps places of chromosomes to avoid situation when they
613  // effectively become two separate populations.
614  HostPopulation.back().swapChromosomes();
615  }
