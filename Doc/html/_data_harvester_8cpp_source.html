46 auto getShannonIndx(std::vector<int> GeneVals){
47  int Types = 0;
48  double Summ = 0.0;
49  double tot_gene_numb = (double) GeneVals.size();
50  if(GeneVals.size()){
51  int GeneCounter = GeneVals.size();
52  bool IfCountedLyst[GeneCounter];
53  for (int w = 0; w < GeneCounter; ++w) {
54  IfCountedLyst[w] = false;
55  }
56  double abundance;
57  for (int i = 0; i < GeneCounter; ++i) {
58  if (IfCountedLyst[i] == false) {
59  Types += 1;
60  abundance = 1.0;
61  for (int j = i + 1; j < GeneCounter - 1; ++j) {
62  if (i != j && IfCountedLyst[j] == false
63  && GeneVals[i] == GeneVals[j]){
64  abundance += 1.0;
65  IfCountedLyst[j] = true;
66  }
67  }
68  // the proper calculations
69  Summ = Summ - ((abundance / tot_gene_numb)
70  * std::log(abundance / tot_gene_numb));
71  }
72  }
73  }
74  // ShannIndx (double), typesOfGenes (int), numberOfAllGenes (double)
75  auto RTRN = std::make_tuple( Summ, Types, tot_gene_numb );
76  return RTRN;
77 }

155  ifError = true;

157  if (HeteroHomo != 10 and HeteroHomo != 11){
158  std::cout<< "\nError in the heterozygote advantage / lack of advantage " <<
159  "mode. It has to be 10 for heterozygote advantage or 11 for " <<
160  "lack of thereof." << std::endl;
161  ifError = true;
162  }
163  if (maxGene < 1.0 or maxGene < hostGeneNumbb){
164  std::cout<< "\nError in the hosts' maximal number of genes per " <<
165  "chromosome. It has to be at least one, but not less then the " <<
166  "number used to initialize the system. "<< std::endl;
167  ifError = true;
168  }
169  return ifError;
170 }

318 void DataHarvester::saveHostGeneticDivers(Environment& EnvObj, int tayme){
319  if(ifFirstHostGeneDivRun){
320  std::ofstream HostGeneDivFile;
321  HostGeneDivFile.open("HostsGeneDivers.csv");
322  HostGeneDivFile << "#time pop_size tot_num_of_genes num_of_MHC_types" <<
323  " Shannon_indx mean_fitness std_fitness" << std::endl;
324  HostGeneDivFile.close();
325  ifFirstHostGeneDivRun = false;
326  }
327  std::vector<int> AllTheGeneVals;
328  std::vector<double> Fitness;
329  double popSize = (double) EnvObj.getHostsPopSize();
330  int homoLociNum = -1; // not applicable at the moment!
331 // double tot_gene_numb = 0;
332  Fitness.clear();
333  for(int i = 0; i < EnvObj.getHostsPopSize(); ++i){
334 // tot_gene_numb += EnvObj.getSingleHostGenomeSize(i);
335  Fitness.push_back(EnvObj.getHostFitness(i));
336  // Harvesting genes from Chromosome One in all hosts
337  for(int j = 0; j < EnvObj.getSingleHostChromoOneSize(i); ++j){
338  AllTheGeneVals.push_back(EnvObj.getSingleHostRealGeneOne(i, j));
339  }
340  // Harvesting genes from Chromosome Two in all hosts
341  for(int m = 0; m < EnvObj.getSingleHostChromoTwoSize(i); ++m){
342  AllTheGeneVals.push_back(EnvObj.getSingleHostRealGeneTwo(i, m));
343  }
344  }
345  // Calculating the Shannon index et al. plus extracting it from tuple
346  auto ShOut = getShannonIndx(AllTheGeneVals);
347  double Summ = std::get < 0 >( ShOut );
348  int mhcTypes = std::get < 1 >( ShOut );
349  double tot_gene_numb = std::get < 2 >( ShOut );
350 
351  // calculating coefficient of variation of hosts fitness
352  double cv_sum = std::accumulate(Fitness.begin(), Fitness.end(), 0.0);
353  double cv_mean = cv_sum / Fitness.size();
354  double sq_sum = std::inner_product(Fitness.begin(), Fitness.end(),
355  Fitness.begin(), 0.0);
356  double stdev = std::sqrt(sq_sum / Fitness.size() - cv_mean * cv_mean);
357 
358  std::ofstream HostGenomesFile;
359  HostGenomesFile.open("HostsGeneDivers.csv",
360  std::ios::out | std::ios::ate | std::ios::app);
361  HostGenomesFile << tayme << " " << popSize << " " << tot_gene_numb << " " <<
362  mhcTypes << " " << Summ << " " << cv_mean <<
363  " " << stdev << std::endl;
364  HostGenomesFile.close();
365 }
