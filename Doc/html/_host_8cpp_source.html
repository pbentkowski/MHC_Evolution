562 void Host::evalUniqueMHCs() {
563  chromovector tmpChrome;
564  int tmpSize, k;
565  tmpChrome.clear();
566  tmpChrome.reserve(ChromosomeOne.size() + ChromosomeTwo.size());
567  tmpChrome.insert(tmpChrome.end(), ChromosomeOne.begin(), ChromosomeOne.end());
568  tmpChrome.insert(tmpChrome.end(), ChromosomeTwo.begin(), ChromosomeTwo.end());
569  tmpSize = (int) tmpChrome.size();
570  k = 0;
571  while(k < tmpSize) {
572  tmpSize = (int) tmpChrome.size();
573  for(int i = tmpSize-1; i > k; --i){
574  if(tmpChrome[k].getTheRealGene() == tmpChrome[i].getTheRealGene()){
575  tmpChrome.erase(tmpChrome.begin() + i);
576  }
577  }
578  k++;
579  }
580  UniqueAlleles = tmpChrome;
581 }
