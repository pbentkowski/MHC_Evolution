515 void Host::evalUniqueMHCs() {
516  chromovector tmpChrome;
517  int tmpSize;
518  tmpChrome.clear();
519  tmpChrome.reserve(ChromosomeOne.size() + ChromosomeTwo.size());
520  tmpChrome.insert(tmpChrome.end(), ChromosomeOne.begin(), ChromosomeOne.end());
521  tmpChrome.insert(tmpChrome.end(), ChromosomeTwo.begin(), ChromosomeTwo.end());
522  int notUniqCount = (int) tmpChrome.size();
523  int k = 0;
524  while(notUniqCount > 0) {
525  notUniqCount = 0;
526  tmpSize = (int) tmpChrome.size();
527  for(int i = tmpSize; i > k; --i){
528  if(tmpChrome[k].getTheRealGene() == tmpChrome[i].getTheRealGene()){
529  tmpChrome.erase(tmpChrome.begin() + i);
530  notUniqCount++;
531  }
532  }
533  k++;
534  }
535  UniqueAlleles = tmpChrome;
536 }
