550 void Host::evalUniqueMHCs() {
551  chromovector tmpChrome;

553  tmpChrome.clear();
554  tmpChrome.reserve(ChromosomeOne.size() + ChromosomeTwo.size());
555  tmpChrome.insert(tmpChrome.end(), ChromosomeOne.begin(), ChromosomeOne.end());
556  tmpChrome.insert(tmpChrome.end(), ChromosomeTwo.begin(), ChromosomeTwo.end());
557  tmpSize = (int) tmpChrome.size();

563  tmpChrome.erase(tmpChrome.begin() + i);
